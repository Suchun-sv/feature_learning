# KMP 

#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
const int M = 1000000+5;
char S[M];
char T[M];
int fail[M];
int m,n;
int main()
{
#ifdef debug
freopen("C:\\Users\\X1 Yoga\\Desktop\\in.txt","r",stdin);
//freopen("C:\\Users\\X1 Yoga\\Desktop\\out1.txt","w",stdout);
#endif
    scanf("%s%s", S+1, T+1);
    n = strlen(S+1), m = strlen(T+1);

    // 构造fail数组 
    for(int i=2;i<=m;i++)
    {
        int j = fail[i-1];
        while(T[i]!=T[j+1]&&j) j = fail[j];
        if(T[i]==T[j+1]) j++;
        fail[i]=j;
    }
    // 开始匹配 
    int p=0;
    for(int i=1;i<=n;i++)
    {
        // 若不匹配，往后移动 
        while(S[i]!=T[p+1]&&p) p =fail[p];
        if(S[i]==T[p+1]) p++;
        if(p==m)
        {
            printf("%d\n",i-m+1);
            p=fail[p];
        }
    }
    for(int i=1;i<=m;i++) printf("%d ", fail[i]);
}
